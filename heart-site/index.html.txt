    display: block;

<body>

<canvas id="canvas"></canvas>

<script>
const canvas = document.getElementById("canvas");
const ctx = canvas.getContext("2d");

canvas.width = window.innerWidth;
canvas.height = window.innerHeight;

let particles = [];
const heartPoints = [];
const TOTAL = 2000;
let mouse = { x: null, y: null };

window.addEventListener("resize", () => {
    canvas.width = window.innerWidth;
    canvas.height = window.innerHeight;
});

window.addEventListener("mousemove", e => {
    mouse.x = e.clientX;
    mouse.y = e.clientY;
});

function heartFunction(t) {
    const x = 16 * Math.pow(Math.sin(t), 3);
    const y =
        13 * Math.cos(t) -
        5 * Math.cos(2 * t) -
        2 * Math.cos(3 * t) -
        Math.cos(4 * t);
    return { x, y };
}

// Create heart shape points
for (let i = 0; i < TOTAL; i++) {
    const t = Math.random() * Math.PI * 2;
    const p = heartFunction(t);
    heartPoints.push({
        x: canvas.width / 2 + p.x * 18,
        y: canvas.height / 2 - p.y * 18
    });
}

// Create particles
for (let i = 0; i < TOTAL; i++) {
    particles.push({
        x: Math.random() * canvas.width,
        y: Math.random() * canvas.height,
        tx: heartPoints[i].x,
        ty: heartPoints[i].y,
        size: Math.random() * 2 + 1,
        speed: Math.random() * 0.02 + 0.01
    });
}

let pulse = 0;

function animate() {
    ctx.fillStyle = "rgba(0,0,0,0.25)";
    ctx.fillRect(0, 0, canvas.width, canvas.height);

    pulse += 0.05;
    const scale = 1 + Math.sin(pulse) * 0.05;

    particles.forEach(p => {
        p.x += (p.tx - p.x) * p.speed;
        p.y += (p.ty - p.y) * p.speed;

        let dx = mouse.x - p.x;
        let dy = mouse.y - p.y;
        let dist = Math.sqrt(dx * dx + dy * dy);

        if (dist < 120) {
            p.x -= dx * 0.03;
            p.y -= dy * 0.03;
        }

        ctx.beginPath();
        ctx.arc(
            canvas.width / 2 + (p.x - canvas.width / 2) * scale,
            canvas.height / 2 + (p.y - canvas.height / 2) * scale,
            p.size,
            0,
            Math.PI * 2
        );
        ctx.fillStyle = "#ff2d55";
        ctx.fill();
    });

    requestAnimationFrame(animate);
}

animate();
</script>
